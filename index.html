// Straight forward jpa domain
@Entity @Table(name="person")
class Person {
	@Id @GeneratedValue(strategy= GenerationType.AUTO) var id:jLng = _
	@Version var version:jInt = _
	var firstName:String = _
	var lastName:String = _
}
// companion 'Data Access Object'
object Person extends JpaDao[Person,jLng]
// controller with standard Create read update delete
class PersonController extends CrudController("persons",Person)
